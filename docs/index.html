Quick Installation

# Clone repository
git clone https://github.com/cppNexus/gas-monitor.git
cd gas-monitor

# Configure environment
cp .env.example .env
nano .env  # Add your Telegram credentials

# Run with Docker
docker-compose up -d

# Or run locally
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt
python main.py
